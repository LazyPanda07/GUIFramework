309 void BaseComboBox::resize(uint16_t width, uint16_t height)
310 {
311 if (autoResize && !blockResize)
312 {
313 LRESULT currentSize = this->size();
314 HDC deviceContext = GetDC(handle);
315 int heightSum = 0;
316
317 if (currentSize == CB_ERR || !currentSize)
318 {
319 double widthCoefficient = static_cast<double>(width) / parentWidth;
320
321 MoveWindow
322 (
323 handle,
324 static_cast<int>(desiredX * widthCoefficient),
325 static_cast<int>(desiredY * (static_cast<double>(height) / parentHeight)),
326 static_cast<int>(desiredWidth * widthCoefficient),
327 desiredHeight,
328 true
329 );
330
331 ShowWindow(handle, SW_SHOW);
332
333 return;
334 }
335
336 for (size_t i = 0; i < currentSize; i++)
337 {
338 wstring value = this->getValue(i);
339 SIZE valueSizes;
340
341 if (GetTextExtentPoint32W(deviceContext, value.data(), value.size(), &valueSizes))
342 {
343 if (valueSizes.cx > requiredSize.cx)
344 {
345 requiredSize.cx = valueSizes.cx;
346 }
347
348 if (valueSizes.cy > requiredSize.cy)
349 {
350 requiredSize.cy = valueSizes.cy;
351 }
352
353 heightSum += valueSizes.cy;
354 }
355 }
356
357 this->setItemHeight(itemHeightEnum::forAllItems, static_cast<uint16_t>(requiredSize.cy));
358
359 MoveWindow
360 (
361 handle,
362 static_cast<int>(desiredX * (static_cast<double>(width) / parentWidth)),
363 static_cast<int>(desiredY * (static_cast<double>(height) / parentHeight)),
364 max(requiredSize.cx + standard_sizes::comboBoxAdditionalWidth, static_cast<long>(desiredWidth * (static_cast<double>(width) / parentWidth))),
365 heightSum + requiredSize.cy * 2,
366 true
367 );
368
369 ShowWindow(handle, SW_SHOW);
370 }
371 }
